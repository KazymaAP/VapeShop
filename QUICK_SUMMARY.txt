### Исправление 3: Redis для CSRF (lib/csrf.ts)

**БЫЛО:**
```typescript
const csrfStore: CSRFStore = {}; // In-memory, утечка памяти!

export function generateCSRFToken(sessionId: string): string {
  const token = crypto.randomBytes(32).toString('hex');
  csrfStore[sessionId] = { token, timestamp: Date.now() }; // Бесконечный рост!
  return token;
}
```

**СТАЛО:**
```typescript
import Redis from 'ioredis';

const redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');
const CSRF_TOKEN_EXPIRY = 24 * 60 * 60; // 24 часа

export async function generateCSRFToken(sessionId: string): Promise<string> {
  const token = crypto.randomBytes(32).toString('hex');
  await redis.setex(`csrf:${sessionId}`, CSRF_TOKEN_EXPIRY, token);
  return token;
}

export async function verifyCSRFToken(sessionId: string, token: string): Promise<boolean> {
  const stored = await redis.get(`csrf:${sessionId}`);
  if (!stored) return false;
  
  // Timing-safe сравнение
  return crypto.timingSafeEqual(
    Buffer.from(stored, 'hex'),
    Buffer.from(token, 'hex')
  );
}
```

### Исправление 4: Rate Limit на Redis (lib/rateLimit.ts)

**БЫЛО:**
```typescript
const store: RateLimitStore = {}; // Race condition!

export function rateLimit(handler, options) {
  return async (req, res) => {
    const key = getKey(req, options.keyGenerator);
    const now = Date.now();
    
    if (!store[key]) {
      store[key] = { count: 1, resetTime: now + options.windowMs };
    } else if (store[key].resetTime <= now) {
      store[key] = { count: 1, resetTime: now + options.windowMs };
    } else {
      store[key].count++; // ← RACE CONDITION!
    }
    
    if (store[key].count > options.max) {
      return res.status(429).json({ error: 'Too many requests' });
    }
    
    return handler(req, res);
  };
}
```

**СТАЛО:**
```typescript
import Redis from 'ioredis';

const redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');

export function rateLimit(handler, options) {
  return async (req, res) => {
    const key = getKey(req, options.keyGenerator);
    
    try {
      // Atomically increment counter
      const count = await redis.incr(`ratelimit:${key}`);
      
      if (count === 1) {
        // Set expiry only on first increment
        await redis.expire(`ratelimit:${key}`, Math.ceil(options.windowMs / 1000));
      }
      
      const resetTime = await redis.ttl(`ratelimit:${key}`);
      
      if (count > options.max) {
        res.setHeader('Retry-After', resetTime);
        return res.status(429).json({ error: 'Too many requests' });
      }
      
      return handler(req, res);
    } catch (err) {
      console.error('Rate limit error:', err);
      return handler(req, res); // Fallback - не блокировать если Redis недоступен
    }
  };
}
```
