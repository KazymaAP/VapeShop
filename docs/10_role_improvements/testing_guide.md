**Пример: Тестирование функции расчёта скидки**

```typescript
// __tests__/lib/discountCalculator.test.ts
import { calculateDiscount } from '../../lib/discountCalculator';

describe('calculateDiscount', () => {
  it('should calculate discount correctly', () => {
    expect(calculateDiscount(1000, 10)).toBe(900); // 1000 - 10% = 900
  });

  it('should handle 0 discount', () => {
    expect(calculateDiscount(1000, 0)).toBe(1000);
  });

  it('should handle 100% discount', () => {
    expect(calculateDiscount(1000, 100)).toBe(0);
  });
});
```
